\begin{frame}[t]{On-demand Summarization}

  \normalsize
\[
\begin{array}{@{}l@{}}
  \mex{(map : (any/c -> any/c) list? -> list?)} \\
  \mex{(\kw{define} (map f xs)} \\
  \hspace{1.2em}\mex{(\kw{cond} ([\altcolor{2}{blue}{(null xs)} \altcolor{3}{blue}{\codequote()}]} \\
  \hspace{5em}\mex{[\kw{else} \altcolor{4}{blue}{(cons (f (car xs)) (map f (cdr xs)))}])))} \\
  \mex{(div : nat? (>/c 0) -> nat?)} \\
  \mex{(main : (listof nat?) -> (listof nat?))} \\
  \mex{(\kw{define} (main ms)} \\
  \hspace{1.2em}\mex{\altcolor{1}{blue}{(map ($\lambda$ (m) (div 42 m)) ms)})} \\
  \\
  \alt<10->{
    \mex{(null ms)} \implies \mex{(map <$\lambda$> ms)} \to \altcolor{10}{blue}{\{\codequote()\}} \\
    \alt<11->{
      \mex{(not (null ms))} \implies \mex{(map <$\lambda$> ms)} \to \altcolor{11}{blue}{\mex{\{<non-empty (listof nat?)>, <blame>\}}}
    }{
      \mex{(not (null ms))} \implies \mex{(map <$\lambda$> ms)} \to \altcolor{10}{blue}{\mex{\{(cons nat? \codequote()), <blame>\}}}
    }
  }{
    \invisible<-2>{\mex{(null ms)} \implies \mex{(map <$\lambda$> ms)} \to \altcolor{3}{blue}{\codequote()}} \\
    \invisible<-3>{
      \alt<7->{
        \mex{(not (null ms))} \implies \mex{(map <$\lambda$> ms)} \to \altcolor{4}{blue}{\mex{(cons \altcolor{7-}{blue}{(div 42 m)} \codequote())}}
      }{
        \alt<6->{
          \mex{(not (null ms))} \implies \mex{(map <$\lambda$> ms)} \to \altcolor{4}{blue}{\mex{(cons (<$\lambda$> m) \altcolor{6}{blue}{\codequote()})}}
        }{
          \mex{(not (null ms))} \implies \mex{(map <$\lambda$> ms)} \to \altcolor{4}{blue}{\mex{(cons (<$\lambda$> m) \altcolor{5}{blue}{(map <$\lambda$> ms')})}}
        }
      }
    }\\
    \invisible<-7>{
      \mex{((and/c nat? (>/c 0)) m)} \implies \mex{(div 42 m)} \to \altcolor{8}{blue}{\mex{nat?}}
    }\\
    \invisible<-8>{
      \mex{((and/c nat? (not (>/c 0))) m)} \implies \mex{(div 42 m)} \to \altcolor{9}{blue}{\mex{<blame>}}

\end{array}
